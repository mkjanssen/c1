\frame{
	\frametitle{Local Linearization}
	
	\begin{itemize}
	\item Equations of lines need a point and slope.\pause
	\item The slope of $f(x)$ at $(a,f(a))$ is $f'(a)$.\pause
	\item The equation of the line tangent to $f(x)$ at $(a,f(a))$ is thus 
	\[
		y = f'(a) (x-a) + f(a)
	\]\pause
	\item Often called the \textbf{\alert{local linearization}} and denoted by $L(x)$
\end{itemize}
	

}

\frame{
	\frametitle{Example}
	
	Let's approximate $\sqrt{4.01}$. \rem{$f'(2) = 1/4$}

}

\frame{
	\frametitle{How good an approximation is the linearization?}
	
	We know $f(a) = L(a)$ and $f'(a) = L'(a)$. What about values of $x$ near $a$?\pause
	
	Consider the second derivative:
	
	\begin{itemize}
	\item $f''(a) > 0$:\pause\ $f$ is concave up, so the tangent line lies entirely below the curve ($L(x) < f(x)$)\pause
	\item $f''(a) < 0$:\pause\ $f$ is concave down, so the tangent line lies entirely above the curve ($L(x) > f(x)$)\pause
	\item $f''(a) = 0$:\pause\ the concavity is changing, and it depends on how (up to down, or down to up)
\end{itemize}

}
