\frame{
	\frametitle{Inverse Functions}
	
	\alert{\textbf{Notation:}} by $f: A\to B$ we mean that $f$ is a function with \emph{domain} $A$ and \emph{codomain} $B$.
	
	\pause
	
	The \textbf{inverse of $f$} (if it exists) is a function $g: B\to A$ such that $g(f(a)) = a$ for every possible choice of $a$ in the domain $A$ and $f(g(b)) = b$ for every possible choice of $b$ in $B$.\pause
	
	We usually denote the inverse of $f$ as $f^{-1}$. Observe that $y = f(x)$ if and only if $x = f^{-1}(y)$.
	
	\pause
	
	\alert{\textbf{Big Important Fact:}} $f: A\to B$ has an inverse if and only if $f$ passes the vertical and horizontal line tests on $A$

}

	\frametitle{Examples of Inverses}
	
	On appropriate domains:
	
	\begin{itemize}
	\item $y = a^x$ if and only if $x = \log_a(y)$
	\item $y = x^n$ if and only if $x = \sqrt[n]{y}$
	\item $y = \sin x$ if and only if $x = \sin^{-1} y = \arcsin y$
	\item etc
\end{itemize}

\frame{
	\frametitle{Coda: the Inverse Function Theorem}
	
	\begin{theorem}[Inverse Functions]
	
	Suppose that $f$ is differentiable with inverse $g$ and that $(x_0,y_0)$ is a point that lies on the graph of $f$ at which $f'(x_0)\ne 0$. Then 
	\[
		g'(y_0) = \frac{1}{f'(x_0)}
	\]
	
	More generally, for any $z$ in the domain of $g'$, we have $g'(z) = 1/f'(g(z))$.

\end{theorem}
