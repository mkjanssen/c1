\frame{
	\frametitle{Meaning of the derivative}
	
	\begin{itemize}
	\item Derivative measures instantaneous rate of change of the output variable with respect to the input variable, regardless of what they represent--doesn't have to be change of position!
	\item \textbf{Units}: units on $y = f'(x)$ are units of $y$ per unit of $x$.
\end{itemize}

}

\frame{
	\frametitle{Example}
	
	Let $W$ be the volume of water, in gallons, in a bathtub at time $t$, in minutes.

\begin{enumerate}[(a)]
	\item What are the meaning and units of $W'(t)$?
	\item Suppose the bathtub is full of water at time $t_0$, so that $W(t_0) > 0$. Subsequently, at time $t_p > t_0$, the plug is pulled. Is $W'(t)$ positive, negative, or zero:

\begin{itemize}
	\item For $t_0 < t < t_p$?
	
		%\vfill
	\item After the plug is pulled, but before the tub is empty?
	
		%\vfill
	\item When all the water has drained from the tub?
	
		%\vfill
\end{itemize}
\end{enumerate}


}

\frame{
	\frametitle{Goal: More accurate estimates of the derivative}
	
	Two choices:
	
	\begin{itemize}
	\item \textbf{Difference Quotient:} $ f'(a) \approx \frac{f(a+h) - f(a)}{h}$\pause
	
	\item \textbf{Central Difference:} $f'(a)\approx\frac{f(a+h) - f(a-h)}{2h}$ \rem{pos/neg $h$'s}\pause
\end{itemize}

The central difference gives a line closer to being parallel to the tangent. \rem{ex: $y = -0.5x^2 + 4$, $a=2$}

}

\frame{
	\frametitle{Example}
	
	A cup of coffee has its temperature $F$ (in degrees F) at time $t$ given by the function $F(t) = 75 + 110e^{-0.05t}$, where time is measured in minutes.
	
	\begin{enumerate}
	\item Use a central difference with $h = 0.01$ to estimate the value of $F'(10)$.\pause
	\item What are the units on the value of $F'(10)$? What is the practical meaning of the value of $F'(10)$?\pause 
	\item Which do you expect to be greater: $F'(10)$ or $F'(20)$? Why?
\end{enumerate}

}
