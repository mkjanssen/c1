\frame{
	\frametitle{Increasing, Decreasing and the Big question}
	
	
	\begin{definition}
Given a function $f(x)$ on an interval $(a,b)$, we say that $f$ is \textbf{increasing on $(a,b)$} provided that for all $x < y$ on $(a,b)$ we have $f(x) < f(y)$. Similarly, we say $f$ is \textbf{decreasing on $(a,b)$} provided that for all $x < y$ on $(a,b)$ we have $f(x) > f(y)$.
\end{definition}
	
	\pause
	
	
	The derivative is a function: how does \emph{it} change? If it's increasing, what does that mean for the original function? \emph{How} is it increasing?
	
	
	
	



}




\frame{
	\frametitle{The Second Derivative}
	
	\begin{definition}
Given a function $f$, the \textbf{second derivative of $f$} is the function
\[
f''(x) = \lim\limits_{h\to 0} \frac{f'(x+h) -f'(x)}{h}
\]
for all $x$ for which the limit exists.
\end{definition}

}

\frame{
	\frametitle{Concavity}
	
	\begin{definition}
Let $f$ be differentiable on $(a,b)$. Then $f$ is \textbf{concave up} on $(a,b)$ if and only if $f'$ is increasing on $(a,b)$. We say $f$ is \textbf{concave down} on $(a,b)$ if and only if $f'$ is decreasing on $(a,b)$.
\end{definition}
