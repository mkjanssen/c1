\frame{
	\frametitle{The Chain Rule}
	
	If $g$ is differentiable at $a$ and $f$ is differentiable at $g(a)$, then the composite function $C$ defined by $C(x) = f(g(x))$ is differentiable at $a$, and 
	\[
		C'(a) = f'(g(a))\cdot g'(a)
	\]

}
