\frame{
	\frametitle{The Connection}
	
	Given a 1-dimensional velocity described by $v(t)$ corresponding to a displacement function $s(t)$, we have seen that the distance traveled between $t=a$ and $t=b$ is given by
	\[
		s(b) - s(a) \pause = \int_a^b v(t)\, dt
	\]
	
	\pause
	
	\begin{theorem}[Fundamental Theorem of Calculus]
	
		If $f$ is continuous on $[a,b]$ and $F$ is any antiderivative of $f$, then
		\[
			\int_a^b f(x)\, dx = F(b) - F(a)\pause = F(x)\biggr\vert_a^b
		\]

\end{theorem}\pause

\alert{The Antiderivative Problem}: given a function $f$, find all antiderivatives $F$ of $f$.
	
	

}

\frame{
	\frametitle{The Total Change Theorem}
	
	In the context of an object moving with velocity $v$, the integral gives the total displacement of the object. That is, it gives the \alert{total change} in the antiderivative.
	
	\pause
	
	\begin{theorem}[Total Change Theorem]
	
	If $f$ is continuously differentiable on $[a,b]$ with derivative $f'$ then
	\[
		f(b) - f(a) = \int_a^b f(x)\, dx.
	\]
	\pause
	That is, the definite integral of the rate of change of a function on $[a,b]$ is the total change of the function itself on $[a,b]$.

\end{theorem}

}
